<p>call your function recursively as:</p>
<pre><code>import os

def print_directory_contents(sPath):
    print sPath
    for dir in os.listdir(sPath):
        if os.path.isdir(os.path.join(sPath, dir)):
            print_directory_contents(os.path.join(sPath, dir))
