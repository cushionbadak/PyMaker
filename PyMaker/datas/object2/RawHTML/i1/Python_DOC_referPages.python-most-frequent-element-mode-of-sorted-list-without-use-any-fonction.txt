<p>my solution with lis being the list operated upon:</p>
<pre><code>counter = 0
my_max = 0
max_index = 0

if len(lis) == 1:
    my_max = 1
    max_index = 0
else:
    for i in range(1,len(lis)):
        if lis[i] == lis[i-1]:
            counter += 1
        else:
            counter = 1
        if counter &gt; my_max:
            my_max = counter
            max_index = i

print(f"occurences: {my_max},mode: {lis[max_index]}")
