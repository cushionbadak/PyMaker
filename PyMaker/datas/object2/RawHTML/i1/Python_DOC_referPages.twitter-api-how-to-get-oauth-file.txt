<p>Try this:</p>
<pre><code>import base64

def oAuth(KEY, SECRET):
    TOKEN_BYTES = bytes(str(KEY) + ':' + str(SECRET), encoding='utf-8')
    AUTH_TOKEN = base64.b64encode(TOKEN_BYTES).decode('utf-8')
    return AUTH_TOKEN

OAUTH_FILE = oAuth(CONSUMER_KEY, CONSUMER_SECRET)
