<p>You can use the <a href="https://docs.python.org/3/library/html.parser.html" rel="nofollow noreferrer"><code>html.parser</code></a></p>
<pre><code>from html.parser import HTMLParser

class BExtractor(HTMLParser):
    def __init__(self, *args, **kwargs):
        super(BExtractor, self).__init__(*args, **kwargs)
        self.is_b = True

    def handle_starttag(self, tag, attrs):
        if tag == "b":
            self.is_b = True

    def handle_endtag(self, tag):
            self.is_b = False

    def handle_data(self, data):
        if self.is_b:
            print(data)

parser = BExtractor()
parser.feed("""html""")
</code></pre>
