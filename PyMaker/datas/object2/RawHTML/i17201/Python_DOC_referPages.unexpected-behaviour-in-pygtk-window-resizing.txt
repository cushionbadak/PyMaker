<div class="post-text" itemprop="text">
<p>You are using a class instead of an instance in two locations, line1 and line2, try gtk.gdk.screen_get_default() instead of gtk.gdk.Screen in both locations.</p>
<pre><code>#!/usr/bin/env python

import gtk

class GettingStarted:
    def __init__(self):
        window = gtk.Window()
        width = gtk.gdk.screen_get_default().get_width()#line1
        height = gtk.gdk.screen_get_default().get_height()#line2
        window.resize(width,height)#line3
        label = gtk.Label("Hello")
        window.add(label)

        window.connect("destroy", lambda q : gtk.main_quit())
        window.show_all()

GettingStarted()
gtk.main()
</code></pre>
</div>
