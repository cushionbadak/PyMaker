<p>Try this:</p>
<pre><code>class class_name:
    def __init__(self, name):
        self.name = name
    def display_name(self):
        return self.name
    def saying(self):
        print("hello %s" % self.name)

first = class_name('Chuck')
second = class_name('Tony')
