<p>You should use <a href="https://docs.python.org/3/library/os.path.html" rel="nofollow noreferrer"><code>os.path.join</code></a> and <code>os.filesep</code> to manipulate filepaths.</p>
<pre><code>import os
import time
def shot():
    ts = time.time()
    path = os.path.join("home","sudhanshu","Desktop","shots","sb")
    extention = ".png"
    screensave = datetime.datetime.fromtimestamp(ts).strftime('%d%m%Y%H%M%S')
    print(os.path.join(path, screensave+extention))
    wd.save_screenshot(os.path.join(path, screensave+extention))
</code></pre>
