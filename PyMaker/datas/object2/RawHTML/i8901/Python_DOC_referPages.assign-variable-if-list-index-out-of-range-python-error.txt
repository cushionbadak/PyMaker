<pre><code>for l1, l2 in zip(open('file1.list'), open ('file2.list')):
 a = 'GGG'
 b = 'CCC'
 try:
  a = fasta1[int(l1)]
  b = fasta2[int(l2)]
 except IndexError:
  pass
</code></pre>
