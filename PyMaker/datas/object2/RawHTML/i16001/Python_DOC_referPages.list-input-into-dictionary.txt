    data = csv.reader(fo)
    for row in data:
        dictionary[row[0]] = row[1]
    fo.close()
    ret = []
    for i in abbrv:
        try:
            ret.append(dictionary[i])
        except KeyError:
            ret.append(i)
    return ret

<pre><code>import csv    
def CreateDictionary(i):
    abbrv = i.lower().split(' ')

    dictionary = {}
    fo = open('filename.csv')
    data = csv.reader(fo)
    for row in data:
        dictionary[row[0]] = row[1]
    fo.close()

    to_return = []
    for c in abbrv:
        to_return.append(dictionary.get(c, c))

    return ' '.join(to_return)  #if you want to return a string, if you want to
                                # return a list, use return to_return
</code></pre>
