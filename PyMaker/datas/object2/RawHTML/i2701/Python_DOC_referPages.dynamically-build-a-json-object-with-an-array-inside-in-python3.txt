<div class="post-text" itemprop="text">
<p>Try:</p>
<pre><code>o ={
    "events":[{"id": item, "name": "blah%s" %(index + 1)}
              for index, item in enumerate(array)],
   "name": "bla"
}

print(o)
# {'events': [{'id': 'a', 'name': 'blah1'}, {'id': 'b', 'name': 'blah2'}, {'id': 'c', 'name': 'blah3'}], 'name': 'bla'}
</code></pre>
