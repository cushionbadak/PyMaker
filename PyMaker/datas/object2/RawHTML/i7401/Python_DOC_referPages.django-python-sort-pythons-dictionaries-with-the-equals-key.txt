<p>Here is the right answer :</p>
<pre><code>&lt;tr&gt;
                  {% for cle, valeur in headersLoop %}
                  {% for cleVD, valeurVD in valuesData %}
                  {% if cle == cleVD %}
                  &lt;td&gt;
                  &lt;p&gt; {{cle}}{{valeurVD}} &lt;/p&gt;
                  &lt;/td&gt;

                  {% endif %}
                  {% endfor %}
                  {% endfor %}
