if classs =='1':
    name = input("What is your name? -  ")

    datafile = '1.txt'
    found = False

    # This is a better way to open the file, closes when done
    with open(datafile, 'r') as f:
        data = f.readlines()
        for line in data:
            if name in line:
                print("True")
                found = True
        if found == False:
            print("False")
