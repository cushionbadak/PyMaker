<p>Since you also ask for a bash solution:</p>
<pre><code>#!/bin/bash

for FILE in *.mrc
do
  NUM=$(echo $FILE | cut -d'_' -f 2)
  mkdir -p Stack_$NUM
  mv $FILE Stack_$NUM
done
</code></pre>
