    while num &gt; 0:
        if num &gt;= 10:

<p>Use the break statement to get out of the loop.</p>
<pre><code>def pvalue(num):
    ans = ''
    while num &gt; 0:
        if 1 &lt;= num &lt;= 9:
            ans += 'B'
            num -= 1
        if num &gt;= 10:
            ans += 'A'
            num -= 10
        if num&lt;=0:
            break
    return ans
</code></pre>
