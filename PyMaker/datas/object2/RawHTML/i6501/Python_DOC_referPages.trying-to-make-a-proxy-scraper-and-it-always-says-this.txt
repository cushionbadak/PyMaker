<p>This is the import logic which works for Python 2.7 and further.</p>
<pre><code>print ("FrogSquad4's Scraper v1.0/ver(https://github.com/djinn)")
import sys
if sys.version_info &gt; (2, 7) and sys.version_info &lt; (3, 0):
    import urllib, urllib2
    import Queue
    import StringIO
else:
    import urllib
    import queue
    from io import StringIO
import time, datetime
import threading
import re
import gzip
import sys
input("Press Enter to Start!")
</code></pre>
