<p><a href="https://www.crummy.com/software/BeautifulSoup/bs4/doc/" rel="nofollow noreferrer">BeautifulSoup</a> is what you are looking for</p>
<pre><code>from bs4 import BeautifulSoup
a='''&lt;h3 class="name"&gt;tamar tamar&lt;/h3&gt;
&lt;ul class="list-inline"&gt;
    &lt;li&gt;gender:female&lt;/li&gt;
    &lt;li&gt;age:20&lt;/li&gt;
&lt;li class="phone" data="0529589055"&gt;phone:  0529589055&lt;/li&gt;
&lt;li class="email" data="tamar0529589055@gmail.com"&gt;email: tamar89055@gmail.com&lt;/li&gt; 
'''
soup = BeautifulSoup(a)
print(soup.find('h3',{"class": "name"}).text)
print(soup.find('li',{"class":'phone'}).text)
</code></pre>
