It's not too hard to implement wikipedia algorithm: </p>
<pre><code>import math


def erf(x):
    n = 1
    res = 0
    res1 = (2 / math.sqrt(math.pi)) * x
    diff = 1
    s = x
    while diff &gt; math.pow(10, -20):
        dividend = math.pow((-1), n) * math.pow(x, 2 * n + 1)
        divider = math.factorial(n) * (2 * n + 1)
        s += dividend / divider
        res = ((2 / math.sqrt(math.pi)) * s)
        diff = abs(res1 - res)
        res1 = res
        n += 1
    return res


print(erf(1))
</code></pre>
