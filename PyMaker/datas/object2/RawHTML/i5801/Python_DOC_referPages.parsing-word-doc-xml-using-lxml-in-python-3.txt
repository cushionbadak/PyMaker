ns_pfx = '{http://schemas.openxmlformats.org/wordprocessingml/2006/main}'
pgSz_el = root.find('.//w:pgSz', ns)
pgMar_el = root.find('.//w:pgMar', ns)

print(pgSz_el.get(ns_pfx + 'w'))           # 12240
print(pgMar_el.get(ns_pfx + 'footer'))     # 720
