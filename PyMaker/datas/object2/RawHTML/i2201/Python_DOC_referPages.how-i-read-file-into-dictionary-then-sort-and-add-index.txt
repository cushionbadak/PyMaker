<pre><code>def readGrades(filename):
    result = dict()
    with open(filename) as f:
        for line in f:
            key, val = line.split(",")
            result[key] = int(val.strip())
    return result

def computeRank(dic):
    rank = sorted(list(dic.values())); rank.reverse()
    for key in dic.keys():
        dic[key] = (dic[key], rank.index(dic[key]))
