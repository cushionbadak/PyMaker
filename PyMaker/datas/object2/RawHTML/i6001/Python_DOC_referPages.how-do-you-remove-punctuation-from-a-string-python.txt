<pre><code>def remove_punctuation(str):
    '''(str) -&gt; str
    Return s with all non-space or non-alphanumeric
    characters removed.

    punctuation = "! ()-[]{};:'\"\,&lt;&gt;./?@#$%^&amp;*_~"
    no_punct = ""
    for c in str:
        if c not in punctuation:

    return(no_punct)
