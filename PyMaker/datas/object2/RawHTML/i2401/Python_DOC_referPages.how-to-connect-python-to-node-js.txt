<p>Here is one way to do it, by spawning a child python process from within node.</p>
<p>test.js:</p>
<pre><code>const { spawn } = require('child_process');
const p = spawn('python', ['yourscript.py', '1']);

p.stdout.on('data', (data) =&gt; {
  console.log(`stdout: ${data}`);
});
</code></pre>
<p>yourscript.py</p>
<pre><code>import sys

def change_var(x):
    print('hello from python: {:d}'.format(int(x) + 2))

change_var(sys.argv[1])
</code></pre>
<p>output of running <code>node ./test.js</code>:</p>
<pre><code>stdout: hello from python: 4
</code></pre>
