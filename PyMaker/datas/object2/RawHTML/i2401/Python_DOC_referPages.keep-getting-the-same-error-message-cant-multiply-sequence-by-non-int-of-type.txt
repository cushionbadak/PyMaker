def find_hforce(F, theta):
    if isinstance(F,float) and isinstance(theta,float):
        Fhorizontal = float(F*cos(theta))
        equationFhorizontal = 'Fcos('+str(theta)+')'
    else:
        Fhorizontal = ''
    return (equationFhorizontal, Fhorizontal)

if __name__ == "__main__":
