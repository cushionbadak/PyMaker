<p>Let's assume that there is a "miles" after every number. (This is an untested code so you can get it edited but I think the idea is right). </p>
<p>EDIT: THIS IS TESTED </p>
<pre><code>import collections
originaldict = {}
newdict = collections.OrderedDict()

def isnum(string):
    try:
        if string is ".":
            return True
        float(string)
        return True

    except Exception:
        return False

for line in open(input_file, "r"):
    string = line[:line.find("miles") - 1]
    print string
    curnum = ""
    for c in reversed(string):
        if not isnum(c):
            break
        curnum = c + curnum
    originaldict[float(curnum)] = []
    originaldict[float(curnum)].append(line)


for num in sorted(originaldict.iterkeys()):
    newdict[num] = []
    newdict[num].append(originaldict[num][0])
    del originaldict[num][0]


with open(output_file, "a") as o:
    for value in newdict.values():
        for line in value:
            o.write(line + "\n")
