<p>Use the <a href="https://docs.python.org/3.0/library/time.html" rel="nofollow noreferrer">time</a> module:</p>
<pre><code>import time

     def printit():
        country_list_id = ['a', 'b', 'c']
        product_list_id = ['1', '2', '3']

        for c in country_list_id:
           for p in product_list_id:
              url = "https://www.example.com/api/country_id=" + c + "&amp;product_id=" + p + "&amp;option="
            test(url)

    def test(url_final):
            get_url = requests.get(url_final)
            get_text = get_url.text
            print(get_text)
            time.sleep(3)


    printit()
</code></pre>
