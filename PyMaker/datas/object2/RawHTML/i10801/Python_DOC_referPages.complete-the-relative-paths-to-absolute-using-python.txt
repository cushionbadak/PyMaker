<p>I found a simpler solution with <a href="https://docs.python.org/3/library/re.html#re.sub" rel="nofollow noreferrer"><code>re.sub</code></a> because it accepts a function as the replace-with parameter.</p>
<pre><code>import re
from urllib.parse import urljoin

abs_url = "https://sample.com/sample-page.html"
my_html = """
    &lt;div class="sample-class"&gt;
        &lt;a href="../new-page.html"&gt;New page&lt;/a&gt;
        &lt;img src="../sample-image.jpg" alt=""&gt;
        &lt;img src="../sample-image2.jpg" alt=""&gt;
    &lt;/div&gt;"""

# "src"
absolutize = lambda m: ' src="' + urljoin(abs_url, m.group(1)) + '"'
my_html = re.sub(r' src="([^"]+)"', absolutize, my_html)
# "href"
absolutize2 = lambda m: ' href="' + urljoin(abs_url, m.group(1)) + '"'
my_html = re.sub(r' href="([^"]+)"', absolutize2, my_html)

    """

        &lt;img src="https://sample.com/sample-image2.jpg" alt=""&gt;
    &lt;/div&gt;"""
