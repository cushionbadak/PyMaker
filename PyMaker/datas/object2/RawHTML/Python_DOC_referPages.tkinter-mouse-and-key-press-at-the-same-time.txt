<p>Not sure if there is an official method to bind both <code>Button-1</code> and <code>Key</code>, but maybe you can work around it.</p>
<pre><code>import tkinter as tk

root = tk.Tk()

tk.Label(root,text="Hi I'm a label").pack()

class Bindings:
    def __init__(self):
        self.but_1 = False
        root.bind("&lt;Button-1&gt;", self.mouse_clicked)
        root.bind("&lt;e&gt;", self.e_clicked)
        root.bind("&lt;ButtonRelease-1&gt;",self.mouse_release)

    def mouse_release(self,event):
        self.but_1 = False

    def mouse_clicked(self,event):
        self.but_1 = True
        print ("Mouse button 1 clicked")

    def e_clicked(self,event):
        if self.but_1:
            print ("Both keys clicked")
            self.but_1 = False
        else:
            print ("Key E pressed")

Bindings()
root.mainloop()
</code></pre>
