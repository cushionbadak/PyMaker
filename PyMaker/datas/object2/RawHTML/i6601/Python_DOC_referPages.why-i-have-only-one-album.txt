<pre><code>def index(request):
    all_albums = Album.objects.all()
    html = []
    for album in all_albums:
        url = '/music/%s/' % album.id
        html.append('&lt;a href ="%s"&gt;%s&lt;/a&gt;&lt;br&gt;' % (url, album.album_title))

<pre><code>def index(request):
    all_albums = Album.objects.all()
    html = []
    for album in all_albums:
        url = '/music/{id}/'.format(id=album.id)
        html.append('&lt;a href ="{url}"&gt;{title}&lt;/a&gt;&lt;br&gt;'.format(url=url, title=album.album_title))
    return HttpResponse(''.join(html))
