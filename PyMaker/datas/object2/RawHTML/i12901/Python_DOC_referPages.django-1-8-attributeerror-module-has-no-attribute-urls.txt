<pre><code>from apps import release_news
</code></pre>
<p>read file <code>apps/release_news/__init__.py</code></p>
<p>I guess this file is empty. Add to this file:</p>
<pre><code>from apps.release_news import urls
</code></pre>
<p>or change <code>DemoGuitarProject/urls.py</code></p>
<pre><code>from apps.release_news import urls as release_news_urls

urlpatterns = [
    url(r'^admin/', include(admin.site.urls)),
    url(r'^release_news/', include(release_news_urls)), 
]
