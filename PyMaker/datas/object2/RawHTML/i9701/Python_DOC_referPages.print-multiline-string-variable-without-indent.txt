<div class="post-text" itemprop="text">
<pre><code>def unindent(string):
    return ''.join(map(str.lstrip, string.splitlines(1)))

# Grid.
grid = """rnbqkbnr
          pppppppp
          ********
          ********
          ********
          PPPPPPPP
          RNBQKBNR"""

# Show grid.
print(unindent(grid))
</code></pre>
